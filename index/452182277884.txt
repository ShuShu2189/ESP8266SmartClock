以下是使用Arduino语言为Esp8266编写WiFi配置程序的示例代码：

```cpp
#include <ESP8266WiFi.h>
#include <EEPROM.h>

// WiFi连接相关参数
const char* ssid = "your_network_ssid";
const char* password = "your_network_password";

// EEPROM存储地址
const int ssidAddress = 0;
const int passwordAddress = 50;

void setup() {
  // 初始化串口
  Serial.begin(115200);

  // 读取EEPROM中保存的WiFi数据
  String storedSSID = readEEPROM(ssidAddress, 32);
  String storedPassword = readEEPROM(passwordAddress, 32);

  // 尝试连接WiFi
  WiFi.begin(storedSSID.c_str(), storedPassword.c_str());

  // 等待WiFi连接
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.println("Connecting to WiFi...");
  }

  // 连接成功
  Serial.println("WiFi connected");
  Serial.print("IP address: ");
  Serial.println(WiFi.localIP());

  // 若连接失败，则以AP模式运行
  if (!isConnected()) {
    startAPMode();
  }
}

void loop() {
  // TODO: 程序主循环
}

// 保存WiFi账号和密码到EEPROM
void saveWiFiCredentials(String ssid, String password) {
  writeEEPROM(ssidAddress, ssid);
  writeEEPROM(passwordAddress, password);
}

// 从EEPROM中读取数据
String readEEPROM(int address, int length) {
  String value = "";
  for (int i = 0; i < length; ++i) {
    char c = EEPROM.read(address + i);
    if (c == '\0') {
      break;
    }
    value += c;
  }
  return value;
}

// 写入数据到EEPROM
void writeEEPROM(int address, String value) {
  for (int i = 0; i < value.length(); ++i) {
    EEPROM.write(address + i, value[i]);
  }
  EEPROM.write(address + value.length(), '\0');
  EEPROM.commit();
}

// 判断是否已连接WiFi
bool isConnected() {
  return WiFi.status() == WL_CONNECTED;
}

// 以AP模式运行
void startAPMode() {
  Serial.println("Starting AP mode");
  
  // 创建热点
  WiFi.softAP("YourAPName", "YourAPPassword");

  // 显示热点IP地址
  Serial.print("AP IP address: ");
  Serial.println(WiFi.softAPIP());

  // TODO: 处理用户连接和设置网页
}
```

请根据您的具体需求修改代码中的网络名称（`ssid`）和网络密码（`password`），以及EEPROM存储地址和热点名称（`YourAPName`）与密码（`YourAPPassword`）等信息。
